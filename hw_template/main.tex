% define the \halfblacksquare command (interpolation between \square and \blacksquare)
\newcommand{\halfblacksquare}{%
  \begin{tikzpicture}[scale=0.25, line width=0.15mm, inner sep=0pt, outer sep=0pt, text depth=0pt]
    % Draw the white square
    \fill[white] (0,0) rectangle (1,1);
    % Draw the black half
    \fill[black] (0,0.5) rectangle (1,1);
    % Draw the border
    \draw (0,0) rectangle (1,1);
  \end{tikzpicture}\hspace{-3pt}
}
